/*

Teclado matricial 4 x 4

      C1   C2   C3   C4
L1   [1]  [2]  [3]  [A]
L2   [4]  [5]  [6]  [B]
L3   [7]  [8]  [9]  [C]
L4   [*]  [0]  [#]  [D]


Ativa linha
Lê colunas
Desativa linha
Vai para próxima linha
*/

#include "driver/gpio.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"

#define ROW1 18
#define ROW2 19
#define ROW3 21
#define ROW4 22

#define COL1 23
#define COL2 5
#define COL3 17
#define COL4 16

int rows[4] = {ROW1, ROW2, ROW3, ROW4};
int cols[4] = {COL1, COL2, COL3, COL4};

char keys[4][4] = {
    {'1','2','3','A'},
    {'4','5','6','B'},
    {'7','8','9','C'},
    {'*','0','#','D'}
};

void app_main(void)
{
    gpio_config_t io_conf = {0};

    // Configurar linhas como saída
    io_conf.mode = GPIO_MODE_OUTPUT;
    io_conf.pin_bit_mask =
        (1ULL<<ROW1) | (1ULL<<ROW2) |
        (1ULL<<ROW3) | (1ULL<<ROW4);
    gpio_config(&io_conf);

    // Configurar colunas como entrada com pull-up
    io_conf.mode = GPIO_MODE_INPUT;
    io_conf.pull_up_en = GPIO_PULLUP_ENABLE;
    io_conf.pin_bit_mask =
        (1ULL<<COL1) | (1ULL<<COL2) |
        (1ULL<<COL3) | (1ULL<<COL4);
    gpio_config(&io_conf);

    while (1)
    {
        for (int i = 0; i < 4; i++)
        {
            // Coloca todas as linhas em nível alto
            for (int j = 0; j < 4; j++)
                gpio_set_level(rows[j], 1);

            // Ativa apenas uma linha
            gpio_set_level(rows[i], 0);

            for (int j = 0; j < 4; j++)
            {
                if (gpio_get_level(cols[j]) == 0)
                {
                    printf("Tecla pressionada: %c\n", keys[i][j]);
                    vTaskDelay(pdMS_TO_TICKS(300)); // debounce simples
                }
            }
        }

        vTaskDelay(pdMS_TO_TICKS(10));
    }
}
