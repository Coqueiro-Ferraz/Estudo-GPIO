#include "driver/gpio.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include <stdio.h>

#define BUTTON_ACTIVE_HIGH 4   // pull-down
#define BUTTON_ACTIVE_LOW  5   // pull-up
#define LED_GPIO           2

void app_main(void)
{
    gpio_config_t io_conf = {0};

    // Configura LED como saída
    io_conf.mode = GPIO_MODE_OUTPUT;
    io_conf.pin_bit_mask = (1ULL << LED_GPIO);
    gpio_config(&io_conf);

    // Configura botão ativo em HIGH (pull-down)
    io_conf.mode = GPIO_MODE_INPUT;
    io_conf.pin_bit_mask = (1ULL << BUTTON_ACTIVE_HIGH);
    io_conf.pull_down_en = GPIO_PULLDOWN_ENABLE;
    io_conf.pull_up_en = GPIO_PULLUP_DISABLE;
    gpio_config(&io_conf);

    // Configura botão ativo em LOW (pull-up)
    io_conf.pin_bit_mask = (1ULL << BUTTON_ACTIVE_LOW);
    io_conf.pull_down_en = GPIO_PULLDOWN_DISABLE;
    io_conf.pull_up_en = GPIO_PULLUP_ENABLE;
    gpio_config(&io_conf);

    while (1)
    {
        int btn_high = gpio_get_level(BUTTON_ACTIVE_HIGH);
        int btn_low  = gpio_get_level(BUTTON_ACTIVE_LOW);

        // Botão ativo em 1
        if (btn_high == 1)
        {
            printf("Botão ativo em HIGH pressionado\n");
            gpio_set_level(LED_GPIO, 1);
        }

        // Botão ativo em 0
        if (btn_low == 0)
        {
            printf("Botão ativo em LOW pressionado\n");
            gpio_set_level(LED_GPIO, 1);
        }

        // Se nenhum pressionado, LED desligado
        if (btn_high == 0 && btn_low == 1)
        {
            gpio_set_level(LED_GPIO, 0);
        }

        vTaskDelay(pdMS_TO_TICKS(100));
    }
}
